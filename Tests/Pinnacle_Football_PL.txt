<script type="text/javascript">
    function fnPlaceItemFirst($scope, element, apply) {
        if (element != null){
            var item = element.parent("li");
            var parent = item.parent("ul");
            if (apply) {
                $scope.$apply(function() {
                    item.prependTo(parent);
                });
            } else {
                item.prependTo(parent);
            }
        }
    }

    function fnInitOddsCombo($scope, $element) {

        var oddSelector = $($element).find(".odds-selector");

        oddSelector.dropdownButton({
            onItemSelected: function() {
                $scope.oddSelected($(this).data('value'));
                window.pinnacle.setUserPrefsCookie("priceStyle", $(this).data('value'));
                fnPlaceItemFirst($scope, $(this), true);
            }
        });
        var cookieValue = window.pinnacle.getUserPrefsCookie("priceStyle");
        var currentItem = null;
        if (!!cookieValue) {
          	cookieValue = cookieValue.toLowerCase();
          	currentItem = oddSelector.find('ul li a[data-value="' + cookieValue + '"]');
        } else {
            var items = oddSelector.find('ul li');
            if (items.length &gt; 0) {
                currentItem = $(items[0]);
            }
        }
        if (currentItem != null){
            currentItem.addClass("active");
            oddSelector.find('.text').text(currentItem.text());
        }      	
        fnPlaceItemFirst($scope, currentItem, false);
        $element.removeClass('hide');
    }
</script>

    function fnPlaceItemFirst($scope, element, apply) {

    function fnInitOddsCombo($scope, $element) {

        var oddSelector = $($element).find(".odds-selector");

        oddSelector.dropdownButton({
            onItemSelected: function() {
                $scope.oddSelected($(this).data('value'));
                window.pinnacle.setUserPrefsCookie("priceStyle", $(this).data('value'));
                fnPlaceItemFirst($scope, $(this), true);
            }
        });
        var cookieValue = window.pinnacle.getUserPrefsCookie("priceStyle");
        var currentItem = null;
        if (!!cookieValue) {
          	cookieValue = cookieValue.toLowerCase();
          	currentItem = oddSelector.find('ul li a[data-value="' + cookieValue + '"]');
        } else {
            var items = oddSelector.find('ul li');
            if (items.length &gt; 0) {
                currentItem = $(items[0]);
            }
        }
        if (currentItem != null){
            currentItem.addClass("active");
            oddSelector.find('.text').text(currentItem.text());
        }      	
        fnPlaceItemFirst($scope, currentItem, false);
        $element.removeClass('hide');
    }
