var currLang;

function changeLang(newLang) {

    var newSearch = location.search;

    if (newSearch == null || newSearch == "") {
        newSearch = "?Lang=" + newLang;
    }
    else {
        newSearch += "&amp;Lang=" + newLang;
    }

    var newPathname = location.pathname;
    var currLangFolder = "/" + currLang.toLowerCase() + "/";

    var idxOfFolderModeLang = location.pathname.toLowerCase().indexOf(currLangFolder);

    if (idxOfFolderModeLang &gt; -1) {
        newPathname = location.pathname.substring(0, idxOfFolderModeLang) + "/" + newLang + "/" + location.pathname.substring(idxOfFolderModeLang + currLangFolder.length);
    }

    var origin = location.protocol + "//" + location.host;

    var newLocation = origin + newPathname + newSearch + location.hash;

    location.href = newLocation;
}

}
